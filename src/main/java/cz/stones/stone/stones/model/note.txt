public class Stone extends AbstractEntity {
    private static final long serialVersionUID = 71442768818150L;

    @NotNull
    private String manufacture;

    @NotNull
    @Enumerated(EnumType.STRING)
    private StateOfStone stateOfRecord = StateOfStone.AVAILABLE;

    @NotNull
    @Column(precision = 4)
    private BigDecimal thicknes;

    @NotNull
    private String color;

    private String notes;

    private String rack;

    @Temporal(TemporalType.TIMESTAMP)
    @NotNull
    private LocalDateTime dateOfCreation;

    @OneToMany(mappedBy = "stone", cascade = CascadeType.PERSIST, fetch = FetchType.LAZY)
    @NotNull
    @Size(min = 1)
    private List<Dimension> dimensions = new ArrayList<>();
}

@Entity
@Getter
@Setter
public class Dimension extends AbstractEntity {

    private static final long serialVersionUID = 71122768818150L;

    @ManyToOne
    @NotNull
    private Stone stone;

    @NotNull
    @Column(precision = 4, scale = 2)
    private BigDecimal dimension;

}



public enum StateOfStone {

    SOLD,
    AVAILABLE;

}
